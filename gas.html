    function pickupId(url) {
        let result = url.replace("https://drive.google.com/file/d/", "");
        result = result.replace("https://drive.google.com/drive/folders/", "");
        result = result.replace("https://docs.google.com/spreadsheets/d/", "");
        result = result.split("?")[0].split("/")[0];
        return result;
    }

    // 「指定フォルダ配下」に絞ることも可能
    function getFolderByName(folderName, rootFolder) {
        let folder = null;
        if (rootFolder) {
            folder = rootFolder.getFoldersByName(folderName);
        } else {
            folder = DriveApp.getFoldersByName(folderName);
        }
        if (folder.hasNext()) {
            return folder.next();
        }
    }

// ============================================
// 共有権限の削除
// ============================================    
    function removeAllShare(folderFile) {
        let editors = folderFile.getEditors();
        for (let i = 0; i < editors.length; i++) {
            folderFile.removeEditor(editors[i].getEmail());
        }
        let viewers = folderFile.getViewers();
        for (let i = 0; i < viewers.length; i++) {
            folderFile.removeViewer(viewers[i].getEmail());
        }
    }

    function addEditors(folderFileId, users) {
        users.forEach(function (mail) {
            Drive.Permissions.create(
            {
                'role': 'writer',
                'type': 'user',
                'emailAddress': mail,
            },
            folderFileId,
            {
                'sendNotificationEmail': 'false',
                'supportsAllDrives': 'true'
            }

        });
    }
